1.exec sp_help lms_book_details;
2.select * from lms_fine_details;
3.select member_name from lms_members where city='chennai';
4.select member_id,member_name,city,membership_status from lms_members where membership_status='permanent';
5.select m.member_id,member_name from lms_members m,lms_book_issue b where m.member_id=b.member_id and b.book_code='BL000002';
6.select book_code,book_title,author from lms_book_details where author like 'P%';
7.select count(*) as 'NO_OF_BOOKS' from lms_book_details group by category having category='java';
8.select category, count(*) as 'NO_OF_BOOKS' from lms_book_details group by category;
9.select count(*) as 'NO_OF_BOOKS' from lms_book_details group by publication having publication='Prentice hall';
10.select bi.book_code,bd.book_title from lms_book_details bd ,lms_book_issue bi where bd.book_code=bi.book_code and date_issue='2012-04-01';

11.select member_id,member_name,date_register,date_expire from lms_members where date_expire<'2013-04-01';
12.select member_id,member_name,date_register,date_expire,membership_status from lms_members where date_register<'2012-03-01' and membership_status='temporary';
13.select member_id,member_name from lms_members where city in ('chennai','delhi');
14.select book_title,'_is_written_by',author from lms_book_details;
15.select avg(price) as 'AVERAGEPRICE' from lms_book_details where category='java';
16.select supplier_id,supplier_name,email from lms_suppliers_details where email like '%gmail%';
17.select supplier_id,supplier_name,coalesce(contact,email,address) as CONTACTDETAILS from lms_suppliers_details;
18.select m.member_id,member_name,bd.book_code,book_title from lms_members m,lms_book_issue bi,lms_book_details bd where m.member_id=bi.member_id and bi.book_code=bd.book_code;
19.select bi.member_id,m.member_name,bi.fine_range,f.fine_amount from lms_members m,lms_fine_details f, lms_book_issue bi where m.member_id=bi.member_id and f.fine_range=bi.fine_range and f.fine_amount<100;
20.select book_code,book_title,rack_num from lms_book_details bd where rack_num='A1' order by book_title;
21.select lms_members.member_id,member_name,date_register from lms_members ,lms_book_issue where lms_members.member_id=lms_book_issue.member_id and date_returned<date_return and date_returned like'2012%';
22.select member_id,member_name,date_register from lms_members where member_id not in (select member_id from lms_book_issue);
23.select  rack_num,count(*) as 'NO_OF_BOOKS' from lms_book_details group by rack_num order by rack_num;
24.select bi.book_issue_no,member_name,date_register,date_expire,book_title,category,author,price,date_issue,date_return,date_returned,'issued'as 'issue_status',fine_amount from lms_members as lm join lms_book_issue as bi on lm.member_id=bi.member_id join lms_book_details as bd on bi.book_code=bd.book_code join lms_fine_details as fd on bi.fine_range=fd.fine_range;
25.select lm.member_id,member_name,3-COUNT(lm.member_id) from lms_book_issue lm,lms_members m WHERE m.member_id=lm.member_id group by lm.member_id,m.member_name;
26.select TOP 3 * from lms_suppliers_details;
27.select * from lms_suppliers_details except (select top 3 * from lms_suppliers_details) ;
28.exec sp_rename 'lms_members','lms_member_details';
29.truncate table lms_fine_details;
30.select all book_title from lms_book_details;
